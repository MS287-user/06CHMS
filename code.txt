const notifRef = useRef(null);

  // Notifications with read/unread status
  const [notifications, setNotifications] = useState([
    { id: 1, message: "New reservation created.", read: false },
    { id: 2, message: "Room 105 ready for check-in.", read: false },
    { id: 3, message: "Guest Ahmed checked out.", read: false },
  ]);

  const [notifOpen, setNotifOpen] = useState(false);

  // Count unread notifications
  const unreadCount = notifications.filter((n) => !n.read).length;

  // Toggle dropdown only
  const handleNotifClick = () => {
    setNotifOpen(!notifOpen);
  };

  // Mark individual notification as read
  const markAsRead = (id) => {
    setNotifications((prev) =>
      prev.map((n) =>
        n.id === id ? { ...n, read: true } : n
      )
    );
  };

  // Close dropdown on outside click
  useEffect(() => {
    const handleOutside = (e) => {
      if (notifRef.current && !notifRef.current.contains(e.target)) {
        setNotifOpen(false);
      }
    };

    document.addEventListener("mousedown", handleOutside);
    return () => document.removeEventListener("mousedown", handleOutside);
  }, []);

{/* NOTIFICATION DROPDOWN */}
        <div className="relative" ref={notifRef}>
          <button onClick={handleNotifClick} className="relative">
            <Bell className="h-6 w-6 text-gray-600 hover:text-blue-600" />

            {/* PILL â€” only show when unread > 0 */}
            {unreadCount > 0 && (
              <span className="absolute -top-1 -right-2 bg-red-500 text-white text-xs w-4 h-4 flex items-center justify-center rounded-full">
                {unreadCount}
              </span>
            )}
          </button>

          {/* DROPDOWN */}
          {notifOpen && (
            <div className="absolute left-0 mt-3 w-72 bg-white border shadow-md rounded-lg py-2 z-50">
              <h3 className="px-4 py-2 text-sm font-semibold text-gray-700 border-b">
                Notifications
              </h3>

              {notifications.map((n) => (
                <div
                  key={n.id}
                  onClick={() => markAsRead(n.id)}
                  className={`px-4 py-2 text-sm cursor-pointer
                  ${n.read ? "text-gray-400 bg-gray-50" : "text-gray-800 hover:bg-gray-100"}
                `}
                >
                  {n.message}
                </div>
              ))}

              {notifications.every((n) => n.read) && (
                <p className="px-4 py-2 text-xs text-gray-500 text-center">
                  No unread notifications
                </p>
              )}
            </div>
          )}
        </div>
